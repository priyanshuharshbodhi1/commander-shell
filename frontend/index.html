Config Showcase

          
One file to rule the fleet.

          
The control plane stays readable. Senior engineers can scan this in seconds, see the topology, and know exactly what will happen when the fleet starts.

        

        

          

            

              
            

            devfleet.yaml / orchestration contract
          

          


# Command the entire engineering fleet from one file
project: devfleet
defaultRuntime: gpt
eventLog: ./.devfleet/events.jsonl

workspaces:
  api:
    provider: docker
    image: node:20-bookworm
  web:
    provider: local
    path: ./apps/web

agents:
  fix-auth:
    runtime: gpt
    workspace: api
    branch: devfleet/fix-auth
    task: Harden session refresh and patch flaky auth tests
  release-notes:
    runtime: claude
    workspace: web
    branch: devfleet/release-notes

plugins:
  tracker: linear
  scm: git
  notifier: slack
  pipeline: github-checks

automation:
  onTaskReady: run: pnpm test
  onChecksGreen: notify: slack
  onMergeReady: openPullRequest: true
